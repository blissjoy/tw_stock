function double_bottom_breakout(stock, t):
    # 偵測兩次分開的打底區間（本函式細節為依圖例歸納後的實作假設，書中無文字演算法）
    bottom1 = detect_bottom_range(stock, t, which="first")
    bottom2 = detect_bottom_range(stock, t, which="second", after=bottom1.end_date)
    if bottom1 is None or bottom2 is None:
        return False

    # 第2次底不創新低
    no_new_low = bottom2.low >= bottom1.low
    if not no_new_low:
        return False

    # 兩底共同壓力區（取兩區間相對高點）
    resistance = max(bottom1.high, bottom2.high)

    is_red_k = Close[t] > Open[t]
    breakout = Close[t] > resistance
    range_avg_vol = average_volume(stock, bottom2.start_date, bottom2.end_date)
    big_volume = Volume[t] >= 2 * range_avg_vol  # 爆大量，沿用成交量分類與倍數門檻定義

    return is_red_k and breakout and big_volume
